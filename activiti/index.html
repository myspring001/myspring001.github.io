查询历史活动流程实例

2023-03-20后端 • 工作流

查询历史活动:

/**
 * 查询历史活动流程实例
 */
@Test
public void queryHistory() {
	HistoryService historyService = engine.getHistoryService();
	
   List<HistoricActivityInstance> list = historyService.createHistoricActivityInstanceQuery().finished()
		   .orderByProcessInstanceId().desc()
		   .list();
   list.forEach(p->{
	   System.out.println("ID："+p.getId());
	   System.out.println("流程实例ID："+p.getProcessInstanceId());
	   System.out.println("活动名称:"+p.getActivityName());
	   System.out.println("办理人："+p.getAssignee());
	   System.out.println("开始时间："+p.getStartTime());
	   System.out.println("结束时间："+p.getEndTime());
	   System.out.println("===================");
   });
}
